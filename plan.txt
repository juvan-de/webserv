client {
	int fd;
	Request request;
	pair<int, string> status;
}

handle_clients(Client client) {
	if POLLIN {
		client.request.addto_request(handle_request(client.getfd()))
	}
	else if POLLOUT {
		handle_response(client.getrequest(), client.getstatus())
	}
}
